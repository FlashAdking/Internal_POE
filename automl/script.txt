# ==========================================
# Windows Commands to Run AutoML Project
# ==========================================

# 1. Open Command Prompt or PowerShell in the 'automl' directory

# 2. Create a virtual environment
python -m venv venv

# 3. Activate the virtual environment
venv\Scripts\activate

# 4. Install dependencies
pip install -r req.txt

# 5. Run the AutoML script
python main.py
